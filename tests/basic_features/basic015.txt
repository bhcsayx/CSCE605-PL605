array[ 10 ] a;

void function outputbackward()
{
    while i > 0 do
        let i := i - 1;
        call OutputNum(a[i]);
    od;
};

procedure outputforward()
{
    while i < 10 do
        call OutputNum(a[i]);
        let i := i + 1;
    od;
};

procedure sort(n)
var i,j,k;
{
    let i := 0;
    while i < n do
        let j := i;
        while j < n do
            if a[i] < a[j] 
            then
                let k := a[i];
                let a[i] := a[j];
                let a[j] := k;
            fi;
            let j := j + 1;
        od;
        let i := i + 1;
    od; 
};
